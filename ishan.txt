#include <iostream>

using namespace std;

 

class PlateDispenser {

private:

    int plates[100];  

    int top;          

    int capacity;

 

public:



    PlateDispenser(int size) {

        capacity = size;

        top = -1;           cout << "Plate Dispenser Initialized with capacity "

             << capacity << endl;

    }

    void addPlate(int plateNumber) {

        if (top == capacity - 1) {

            cout << "Dispenser is FULL! Cannot add more plates.\n";

        } else {

            plates[++top] = plateNumber;

            cout << "Plate " << plateNumber

                 << " added to dispenser.\n";

        }

    }

    void takePlate() {

        if (top == -1) {

            cout << "No plates available!\n";

        } else {

            cout << "Plate " << plates[top]

                 << " taken by customer.\n";

            top--;

        }

    }

 

    void plateCount() {

        cout << "Total plates in dispenser: "

             << (top + 1) << endl;

    }

    void displayPlates() {

        if (top == -1) {

            cout << "Dispenser is empty.\n";

        } else {

            cout << "Plates in dispenser (top to bottom):\n";

            for (int i = top; i >= 0; i--) {

                cout << plates[i] << endl;

            }

        }

    }

 PlateDispenser() {

        cout << "Plate Dispenser system shutting down.\n";

    }

};

int main() {

PlateDispenser dispenser(5);int choice, plateNumber;

do {

        cout << "\n---- Cafeteria Plate Dispenser ----\n";

        cout << "1. Add Clean Plate\n";

        cout << "2. Take Plate\n";

        cout << "3. Check Plate Count\n";

        cout << "4. Display Plates\n";

        cout << "5. Exit\n";

        cout << "Enter choice: ";

        cin >> choice;

 

    switch (choice) {

 

            case 1:

                cout << "Enter Plate Number: ";

                cin >> plateNumber;

                dispenser.addPlate(plateNumber);

                break;

 

            case 2:

                dispenser.takePlate();

                break;

 

            case 3:

                dispenser.plateCount();

                break;

 

            case 4:

                dispenser.displayPlates();

                break;

 

            case 5:

                cout << "Exiting system...\n";

                break;

 

            default:

                cout << "Invalid choice!\n";

        }

 

    } while (choice != 5);

 

    return 0;

}
