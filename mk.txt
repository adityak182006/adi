CREATEORREPLACE PROCEDURE increase_salary
IS
BEGIN--Increasesalaryby10%
UPDATEEMPLOYEE38
SETemp_salary=emp_salary + (emp_salary * 0.10); 
COMMIT;--Displayupdatedsalaries 
FORrecIN(SELECT emp_id, emp_name, emp_salary FROM EMPLOYEE38) 
LOOP
DBMS_OUTPUT.PUT_LINE( 
'ID:'||rec.emp_id || 
'Name:'||rec.emp_name || 
'UpdatedSalary: ' || rec.emp_salary 
);
END LOOP;
END;
/



Function: 

CREATEORREPLACE FUNCTION emp_avg_salary
RETURNNUMBER
IS
v_avg_salaryNUMBER; 
BEGIN--Calculateaverage salary
SELECTAVG(emp_salary)
INTOv_avg_salary
FROMEMPLOYEE38; 
DBMS_OUTPUT.PUT_LINE('Average Salary: ' || v_avg_salary); 
DBMS_OUTPUT.PUT_LINE('Employees having salary greater than average:'); --Displayemployees with salary greater than average 
FOR rec IN (
SELECTemp_id, emp_name, emp_dept, emp_salary 
FROMEMPLOYEE38 
WHEREemp_salary > v_avg_salary 
)
LOOP
DBMS_OUTPUT.PUT_LINE( 
'ID:'||rec.emp_id || 
'Name:'||rec.emp_name || 
'Dept:'||rec.emp_dept || 
'Salary:'||rec.emp_salary 
);
END LOOP;
RETURNv_avg_salary; 
END;
/

cursor:

DECLARE
CURSOR new_cursor IS
SELECT rno, name FROM new_numvb;
CURSOR check_cursor(p_rno NUMBER) IS
SELECT rno
FROM old_numvb
WHERE rno = p_rno;
v_rno old_numvb.rno%TYPE;
BEGIN
FOR rec IN new_cursor LOOP
OPEN check_cursor(rec.rno);
FETCH check_cursor INTO v_rno;
IF check_cursor%NOTFOUND THEN
INSERT INTO old_numvb (rno, name)
VALUES (rec.rno, rec.name);
END IF;
CLOSE check_cursor;
END LOOP;
COMMIT;
DBMS_OUTPUT.PUT_LINE('Data merged successfully');
END;
/
