package ass1;
import java.util.Scanner;
public class assingment2 {
public static String toTitleCase(String name) {
String[] words = name.toLowerCase().split(" ");
StringBuilder titleCase = new StringBuilder();
for (String word : words) {
if (!word.isEmpty()) {
titleCase.append(Character.toUpperCase(word.charAt(0)))

.append(word.substring(1))
.append(" ");

}
}
return titleCase.toString().trim();
}
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
System.out.print("Enter student's full name: ");
String fullName = sc.nextLine();
System.out.print("Enter middle name to search: ");
String middleName = sc.nextLine();

if (fullName.contains(middleName)) {
System.out.println("Middle name found.");
} else {
System.out.println("Middle name not found.");
}

System.out.print("Enter word/character to replace: ");
String oldValue = sc.nextLine();

System.out.print("Enter new word/character: ");
String newValue = sc.nextLine();

String replacedName = fullName.replace(oldValue, newValue);
System.out.println("Name after replacement: " + replacedName);

String titleCaseName = toTitleCase(replacedName);
System.out.println("Name in Title Case: " + titleCaseName);

sc.close();
}
}

import java.util.Scanner;

class Vehicle {
    String brand;
    String model;

    Vehicle(String brand, String model) {
        this.brand = brand;
        this.model = model;
    }

    void displayVehicleInfo() {
        System.out.println("Brand: " + brand);
        System.out.println("Model: " + model);
    }
}

class Car extends Vehicle {
    String fuelType;
    double mileage;

    Car(String brand, String model, String fuelType, double mileage) {
        super(brand, model);
        this.fuelType = fuelType;
        this.mileage = mileage;
    }

    void displayCarInfo() {
        displayVehicleInfo();
        System.out.println("Fuel Type: " + fuelType);
        System.out.println("Mileage: " + mileage + " km/l");
    }
}

class Bike extends Vehicle {
    int engineCapacity;
    String bikeType;
    double mileage;

    Bike(String brand, String model, int engineCapacity, String bikeType, double mileage) {
        super(brand, model);
        this.engineCapacity = engineCapacity;
        this.bikeType = bikeType;
        this.mileage = mileage;
    }

    void displayBikeInfo() {
        displayVehicleInfo();
        System.out.println("Engine Capacity: " + engineCapacity + " cc");
        System.out.println("Bike Type: " + bikeType);
        System.out.println("Mileage: " + mileage + " km/l");
    }
}

public class MainVehicle {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        // Car Input
        System.out.print("Enter Car Brand: ");
        String carBrand = sc.nextLine();

        System.out.print("Enter Car Model: ");
        String carModel = sc.nextLine();

        System.out.print("Enter Fuel Type: ");
        String fuelType = sc.nextLine();

        System.out.print("Enter Car Mileage: ");
        double carMileage = sc.nextDouble();
        sc.nextLine();

        Car car1 = new Car(carBrand, carModel, fuelType, carMileage);

        // Bike Input
        System.out.print("Enter Bike Brand: ");
        String bikeBrand = sc.nextLine();

        System.out.print("Enter Bike Model: ");
        String bikeModel = sc.nextLine();

        System.out.print("Enter Engine Capacity: ");
        int engineCapacity = sc.nextInt();
        sc.nextLine();

        System.out.print("Enter Bike Type: ");
        String bikeType = sc.nextLine();

        System.out.print("Enter Bike Mileage: ");
        double bikeMileage = sc.nextDouble();

        Bike bike1 = new Bike(bikeBrand, bikeModel, engineCapacity, bikeType, bikeMileage);

        System.out.println("\nCar Details:");
        car1.displayCarInfo();

        System.out.println("\nBike Details:");
        bike1.displayBikeInfo();

        sc.close();
    }
}

import java.util.*;

class Document {
    String File;

    Document(String File) {
        this.File = File;
    }

    void show() {
        System.out.println("FILE PRINTED = " + File);
    }
}

class Pdf extends Document {

    Pdf(String File) {
        super(File);
    }

    void show() {
        System.out.println("PDF FILE PRINTED = " + File);
    }
}

class Excel extends Document {

    Excel(String File) {
        super(File);
    }

    void show() {
        System.out.println("Excel FILE PRINTED = " + File);
    }
}

class Word extends Document {

    Word(String File) {
        super(File);
    }

    void show() {
        System.out.println("Word FILE PRINTED = " + File);
    }
}

public class MainDocument {

    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);

        System.out.println("ENTER PDF FILE TO BE PRINTED = ");
        String pdf = sc.nextLine();

        System.out.println("ENTER EXCEL FILE TO BE PRINTED = ");
        String ex = sc.nextLine();

        System.out.println("ENTER WORD FILE TO BE PRINTED = ");
        String w = sc.nextLine();

        Document d = new Pdf(pdf);
        Document d1 = new Excel(ex);
        Document d2 = new Word(w);

        d.show();
        d1.show();
        d2.show();
    }
}
